// ============================================================
// ESP32 #1 — Sensor Node
// PAJ7620U2 via RevEng_PAJ7620 library
// Sensor mounted rotated 90° clockwise — gestures remapped
// Buzzer module: S → GPIO32, VCC → 3.3V, − → GND
// ============================================================

#include <Wire.h>
#include <WiFi.h>
#include <esp_now.h>
#include <esp_wifi.h>
#include "RevEng_PAJ7620.h"

RevEng_PAJ7620 sensor = RevEng_PAJ7620();

// ---- Pins ----
#define LED_RED    25
#define LED_GREEN  18
#define LED_BLUE   27
#define LED_YELLOW 14
#define BUZZER     32

// ---- ESP-NOW ----
uint8_t logicNodeMAC[] = {0x94, 0xB9, 0x7E, 0xD6, 0x4C, 0x00};

// ---- Gesture codes ----
#define G_NONE     0
#define G_UP       1
#define G_DOWN     2
#define G_LEFT     3
#define G_RIGHT    4
#define G_FORWARD  5
#define G_BACKWARD 6
#define G_CW       7
#define G_CCW      8

// ---- Message struct ----
typedef struct {
  uint8_t type;
  uint8_t gesture;
  uint8_t cmd;
} Message;

// ---- State ----
enum State { IDLE, CREATE_MODE, AUTH_MODE };
volatile State currentState = IDLE;

// Trigger sequence: RIGHT, RIGHT, LEFT
const uint8_t TRIGGER[] = {G_RIGHT, G_RIGHT, G_LEFT};
#define TRIGGER_LEN 3
uint8_t triggerIdx = 0;

#define AUTH_WAKE G_UP

uint8_t sequenceCount = 0;
volatile uint8_t lastAck = 0;

// ---- Flags for server-triggered mode changes ----
volatile bool enterAuthFromServer   = false;
volatile bool enterCreateFromServer = false;

// ---- ESP-NOW receive callback ----
void onReceive(const esp_now_recv_info_t *info, const uint8_t *data, int len) {
  Serial.printf("[Sensor RX] len=%d", len);
  if (len >= 1) {
    uint8_t code = data[0];
    Serial.printf(" code=%d state=%d\n", code, (int)currentState);
    if (code == 0 && currentState == IDLE)      enterCreateFromServer = true;
    else if (code == 1 && currentState == IDLE) enterAuthFromServer = true;
    else if (code >= 2)                         lastAck = code;
  } else {
    Serial.println();
  }
}

// ---- Gesture reading with rotation remap ----
uint8_t readGesture() {
  Gesture raw = sensor.readGesture();

  // Sensor rotated 90° clockwise — remap accordingly:
  // Physical DOWN  → logical RIGHT
  // Physical UP    → logical LEFT
  // Physical LEFT  → logical DOWN
  // Physical RIGHT → logical UP
  switch (raw) {
    case GES_UP:            return G_LEFT;
    case GES_DOWN:          return G_RIGHT;
    case GES_LEFT:          return G_DOWN;
    case GES_RIGHT:         return G_UP;
    case GES_FORWARD:       return G_FORWARD;
    case GES_BACKWARD:      return G_BACKWARD;
    case GES_CLOCKWISE:     return G_CW;
    case GES_ANTICLOCKWISE: return G_CCW;
    default:                return G_NONE;
  }
}

// ---- LED / Buzzer helpers ----
void allLedsOff() {
  digitalWrite(LED_RED, LOW); digitalWrite(LED_GREEN, LOW);
  digitalWrite(LED_BLUE, LOW); digitalWrite(LED_YELLOW, LOW);
}
void buzzerOn()  { digitalWrite(BUZZER, HIGH); }
void buzzerOff() { digitalWrite(BUZZER, LOW);  }
void beep(int ms) {
  buzzerOn(); delay(ms); buzzerOff();
}
void blinkLed(int pin, int times, int ms) {
  for (int i = 0; i < times; i++) {
    digitalWrite(pin, HIGH); delay(ms);
    digitalWrite(pin, LOW);  delay(ms);
  }
}
void feedbackGestureReceived() {
  blinkLed(LED_YELLOW, 1, 80);
  beep(60);
}
void feedbackCreateMode() {
  allLedsOff();
  blinkLed(LED_BLUE, 3, 100);
  digitalWrite(LED_BLUE, HIGH);
}
void feedbackSaved() {
  allLedsOff();
  digitalWrite(LED_GREEN, HIGH);
  beep(400);
  delay(800);
  digitalWrite(LED_GREEN, LOW);
}
void feedbackAuthMode() {
  allLedsOff();
  digitalWrite(LED_BLUE, HIGH);
}
void feedbackAuthSuccess() {
  allLedsOff();
  blinkLed(LED_GREEN, 3, 150);
  beep(100); delay(80); beep(100);
}
void feedbackAuthFail() {
  allLedsOff();
  digitalWrite(LED_RED, HIGH);
  buzzerOn(); delay(800); buzzerOff();
  delay(600);
  digitalWrite(LED_RED, LOW);
}

// ---- ESP-NOW send ----
void sendMessage(uint8_t type, uint8_t gesture, uint8_t cmd) {
  Message msg = {type, gesture, cmd};
  esp_now_send(logicNodeMAC, (uint8_t*)&msg, sizeof(msg));
}

// ---- Setup ----
void setup() {
  Serial.begin(115200);
  delay(3000);
  Serial.println("--- Sensor Node Booting ---");

  Wire.begin();
  if (!sensor.begin()) {
    Serial.println("ERROR: PAJ7620 not found! Check wiring.");
    while (true) delay(1000);
  }
  Serial.println("PAJ7620 initialized");

  pinMode(LED_RED, OUTPUT); pinMode(LED_GREEN, OUTPUT);
  pinMode(LED_BLUE, OUTPUT); pinMode(LED_YELLOW, OUTPUT);
  pinMode(BUZZER, OUTPUT);
  allLedsOff();
  buzzerOff();
  Serial.println("Pins initialized");

  WiFi.mode(WIFI_STA);
  WiFi.disconnect();  // don't actually connect to anything

  // Force to match Logic Node's WiFi channel (change 6 to your actual channel)
  esp_wifi_set_channel(6, WIFI_SECOND_CHAN_NONE);
  Serial.printf("Sensor MAC: %s\n", WiFi.macAddress().c_str());

  if (esp_now_init() != ESP_OK) {
    Serial.println("ERROR: esp_now_init failed!");
    while (true) delay(1000);
  }
  Serial.println("ESP-NOW initialized");

  esp_now_register_recv_cb(onReceive);

  esp_now_peer_info_t peer = {};
  memcpy(peer.peer_addr, logicNodeMAC, 6);
  peer.channel = 6;
  peer.encrypt = false;

  if (esp_now_add_peer(&peer) != ESP_OK) {
    Serial.println("ERROR: Failed to add peer!");
    while (true) delay(1000);
  }

  Serial.println("Sensor Node ready.");
}

// ---- Main loop ----
void loop() {
  // ---- Check for server-triggered mode changes ----
  if (currentState == IDLE && enterAuthFromServer) {
    enterAuthFromServer = false;
    sequenceCount = 0;
    currentState = AUTH_MODE;
    feedbackAuthMode();
    Serial.println("Entering AUTH MODE (server command)");
  }
  if (currentState == IDLE && enterCreateFromServer) {
    enterCreateFromServer = false;
    sequenceCount = 0;
    currentState = CREATE_MODE;
    feedbackCreateMode();
    Serial.println("Entering CREATE MODE (server command)");
  }

  uint8_t g = readGesture();

  if (g == G_NONE) { delay(50); return; }

  // Print the logical gesture name for debugging
  const char* names[] = {"NONE","UP","DOWN","LEFT","RIGHT","FORWARD","BACKWARD","CW","CCW"};
  Serial.printf("Gesture: %s | State: %d\n", names[g], currentState);
  delay(300);

  switch (currentState) {

    case IDLE:
      if (g == TRIGGER[triggerIdx]) {
        triggerIdx++;
        feedbackGestureReceived();
        if (triggerIdx == TRIGGER_LEN) {
          triggerIdx = 0;
          sequenceCount = 0;
          currentState = CREATE_MODE;
          feedbackCreateMode();
          sendMessage(0, 0, 0);
          Serial.println("Entering CREATE MODE");
        }
      } else {
        triggerIdx = 0;
        if (g == AUTH_WAKE) {
          sequenceCount = 0;
          currentState = AUTH_MODE;
          feedbackAuthMode();
          sendMessage(0, 0, 1);
          Serial.println("Entering AUTH MODE");
        }
      }
      break;

    case CREATE_MODE:
      feedbackGestureReceived();
      sendMessage(0, g, 2);
      sequenceCount++;
      Serial.printf("Create: gesture %d/6\n", sequenceCount);
      if (sequenceCount == 6) {
        delay(200);
        feedbackSaved();
        currentState = IDLE;
        enterCreateFromServer = false;
        enterAuthFromServer = false;
        allLedsOff();
      }
      break;

    case AUTH_MODE:
      feedbackGestureReceived();
      sendMessage(0, g, 2);
      sequenceCount++;
      Serial.printf("Auth: gesture %d/6\n", sequenceCount);
      if (sequenceCount == 6) {
        delay(300);
        unsigned long t = millis();
        while (lastAck == 0 && millis() - t < 1000) delay(10);
        if (lastAck == 2) feedbackAuthSuccess();
        else              feedbackAuthFail();
        lastAck = 0;
        currentState = IDLE;
        enterCreateFromServer = false;
        enterAuthFromServer = false;
        allLedsOff();
      }
      break;
  }
}
